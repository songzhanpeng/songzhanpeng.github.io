# 配置 react

# 一. 使用 pnpm 创建 react-ts 项目

# pnpm
npm i pnpm -g

# vite
pnpm create vite

√ Project name: ... react-study
√ Select a framework: » react
√ Select a variant: » react-ts

cd react-study
pnpm install
pnpm run dev

# 二. 使用路由

# 1. 安装 react-router-dom

# pnpm
pnpm i react-router-dom -D

# 2. 使用

TIP

App 外层必须包裹 BrowserRouter 或者 HashRouter

// main.tsx
import React from 'react'
import ReactDOM from 'react-dom/client'
import App from './App'
import './index.css'
// 包裹 App
import { BrowserRouter } from 'react-router-dom'

ReactDOM.createRoot(document.getElementById('root')!).render(
  <BrowserRouter>
    <App />
  </BrowserRouter>
)

// App.tsx
// NavLink 跳转组件
// Route 路由组件
// Routes 包裹路由组件
import { NavLink, Route, Routes } from "react-router-dom";

function App() {

  return (
    <div>
      <div className="nav">
        <NavLink to={'/camera'}>摄像机</NavLink>
        <NavLink to={'/mac'}>mac</NavLink>
      </div>
      <Routes> {/* Switch */}
        <Route path="/camera" element={<div>摄像机</div>}></Route>
        <Route path="/mac" element={<div>mac</div>}></Route>
      </Routes>
    </div>
  );
}
export default App;

# 3. 编程式跳转

TIP

点击查看详细 useNavigateopen in new window

// App.tsx
// useNavigate 跳转钩子函数
import { NavLink, Route, Routes, useNavigate } from "react-router-dom";

function App() {
    const navigate = useNavigate()

    const goPage = (path:string) => {
        navigate(path)
    }

  return (
    <div>
      <div className="nav">
        <div onClick={() => goPage('/camera')}>摄像机</div>
        <div onClick={() => goPage('/camera')}>mac</div>
      </div>
      <Routes> {/* Switch */}
        <Route path="/camera" element={<div>摄像机</div>}></Route>
        <Route path="/mac" element={<div>mac</div>}></Route>
      </Routes>
    </div>
  );
}
export default App;

# 三、 进阶路由封装

# 1. 创建 index.tsx

// 文件 src/route/index.tsx
import { useRoutes } from "react-router-dom";

import Home from "@/view/Home";
import Login from "@/view/Login";
import Home1 from "@/view/Home1"
import Home2 from "@/view/Home2"

// 嵌套路由
const RouterView = () => {
  const routes = useRoutes([
    {
      path: "/",
      element: <Login />,
    },
    {
      path: "/home",
      element: <Home />,
      children: [
        {
          path: "home1",
          element: <Home1 />,
        },
        {
          path: "home2",
          element: <Home2 />,
        },
      ],
    },
  ]);
  return routes;
};

export default RouterView;

# 2. 使用封装好的 RouterView

# (1). 第一层路由使用

// src/App.tsx
import { NavLink } from "react-router-dom";

import "./App.css";
import { RouterView } from "./route";

function App() {
  return (
    <div>
      <div className="nav">
        <NavLink to={'/'}>login</NavLink>
        <NavLink to={'/home'}>home</NavLink>
      </div>
      {/* RouterView */}
      <RouterView></RouterView>
    </div>
  );
}
export default App;

# (2). 第二层路由使用

TIP

Outlet 类似于占位符 useParams router6 钩子函数为了取参数

点此查看详情 Outletopen in new window

点此查看详情 useParamsopen in new window

// src/view/Home.tsx
import React from 'react'
import { Outlet, useParams } from 'react-router-dom'

function Home() {
  const {id} = useParams()
  return (
    <div>
      home {id}
      <Outlet></Outlet>
    </div>
  )
}

export default Home

# 四、 配置别名

# 1. vite.config.ts

TIP

需要安装 @types/node , 不然报警告

import { resolve } from 'path'

import { defineConfig } from 'vite'
import react from '@vitejs/plugin-react'
import { resolve } from 'path'

// https://vitejs.dev/config/
export default defineConfig({
  plugins: [react()],
  resolve: {
    alias: {
      '@': resolve(__dirname, './src')
    }
  }
})
